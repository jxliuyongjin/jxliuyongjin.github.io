# 8、Promise异步

函数前面需要加上：async

methods: {
//promise链式调用
this.getNav().then(res=>{
  let id=res.data[0].id;
  return this.getArticle(id);
}).then(res=>{
  let id=res.data[0].id;
  return this.getComment(id)
}).then(res=>{
  console.log(res)
})  

//先获取导航分类接口，将结果进行返回，到调用函数的地方获取
getNav(callback){
  return new Promise((resolve,reject)=>{
    uni.request({
      url:"https://ku.qingnian8.com/dataApi/news/navlist.php",
      success:res=>{
        resolve(res)
      },
      fail:err=>{
        reject(err)
      }
    })
  })
}, 
//获取文章数据，将文章列表进行返回
getArticle(id){
  return new Promise((resolve,reject)=>{
    uni.request({
      url:"https://ku.qingnian8.com/dataApi/news/newslist.php",
      data:{
        cid:id
      },
      success:res=>{
        resolve(res)
      },
      fail:err=>{
        reject(err)
      }
    })
  })
},

//获取文章下的所有评论
getComment(id){
  return new Promise((resolve,reject)=>{
    uni.request({
      url:"https://ku.qingnian8.com/dataApi/news/comment.php",
      data:{
        aid:id
      },
      success:res=>{
        resolve(res)
      },
      fail:err=>{
        reject(err)
      }
    })
})

# 二、组件

# 1、创建

根目录下创建文件夹 components-->创建组件文件夹headCom-->右击创建组件headcomponent

# 2、引用

2.1、在需要引用界面所在的json里面配置引用

"usingComponents": {
  "headerCom":"/components/headCom/headcomponent"
}

2.2、 wxml里面直接使用

<headerCom></headerCom>

# 3、定制参数

3.1、配置组件属性

在组件的js文件夹中配置属性内容

/**
 * 组件的属性列表
 */
properties: {
  titleStr:
  {
    type:String,
    value:"标题",
  },
  content:
  {
    type:String,
    value:"内容内容内容内容内容"
  }
},

3.2、使用组件属性

<view class="header">
  <view class="big">
    {{titleStr}}
  </view> 
  <view class="small">
    {{content}}
  </view>
</view>

3.3、传递参数

引用界面可以传递自己的参数了

<headerCom titleStr="首页Item1" content="首页内容11111"></headerCom>

# 三、使用sass

1、vscode中安装easysass

2、导入已安装的vscode扩展

微信开发者工具 视图=>扩展=>右侧三点=>导入已安装的vscode扩展

3、微信开发者工具 编辑=>打开编辑器扩展目录=>进入sass=>编辑package.json文件=>内容如下

"easysass.formats": {
					"type": "array",
					"default": [
						{
							"format": "expanded",
							"extension": ".wxss"
						} 
					],
					"description": "Define format(s) for outputted css files. Use \"nested\", \"expanded\", \"compact\" or \"compressed\" as a format."
				},

4、重新启动，微信开发者工具

5、使用

​ 5.1、在界面文件夹下创建index.scss

​ 5.2、编辑scss文件，就会自动编写对应的wxcss

例：

.header{
  padding: 10rpx;
  text-align: center; 
  background-color: green;
  .big{ 
    font-size: 40rpx;
    margin-bottom: 10rpx;
  }
  .small{
    font-size: 30rpx;
  }
}
