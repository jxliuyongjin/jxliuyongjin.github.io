# 8、Promise异步

# 8.1、链式调用

test1fun().then(res=>{
    console.log(res); 
    //这里执行test2fun的时候没有成功，处于reject状态拒绝往下执行了 
    return test2fun();
}).then(res=>{ 
    //这里执行test3fun的时候没有成功，抛出异常了，也不会往下执行了
    return test3fun();
    console.log(res)
}).then(res=>{
    return test4fun();
    console.log(res)
}).catch(err=>{ 
    console.log("catch:"+err)
}).finally(()=>{ 
    console.log("finally out") 
}) 

var test1fun = function(){
    return  new Promise((resolve,reject)=>{
    setTimeout(() => {
        resolve("test1 out");
    }, 1000);
})} 


var test2fun = function(){
    return  new Promise((resolve,reject)=>{
    setTimeout(() => {
        reject("test2 reject out");
    }, 1000);
})} 


var test3fun = function(){
    return new Promise((resolve,reject)=>{
    setTimeout(() => {
        reject(new Error("test3 throw Error")); 
    }, 1000);
})} 


var test4fun = function(){
    return new Promise((resolve,reject)=>{
    setTimeout(() => {
        resolve("test4 out");
    }, 1000);
})}

# 8.2、await和promise


//Promise里面执行reject的情况，reject状态，await 调用会直接报错
var test2fun =  new Promise((resolve,reject)=>{
    setTimeout(() => { 
        reject("test2 reject out");  
    }, 1000);
})

///方法1：这种方法可以
var run = async function()
{  
    try {
        //如果不用trycatch,会报错：Uncaught (in promise) test2 reject out
        var rej =  await test2fun; 
        //这里没法执行，会报错
        console.log("rej:"+rej)
    } catch (error) {
        console.log("out error"+error);
    }
}

///方法2：Promise自身的catch
var run = async function()
{   
    //如果不用trycatch,会报错：Uncaught (in promise) test2 reject out
    var rej =  await test2fun.catch(err=>err); 
    //这里没法执行，会报错
    console.log("rej:"+rej) 
}

///方法2：同时获取成功和失败的值
var run = async function()
{   
    //同时获取成功和失败的值
    var [res,err] =  await test2fun.then(res=>[res,null]).catch(err=>[null,err]); 
    //这里没法执行，会报错
    console.log("res:"+res);
    console.log("err:"+err); 
}


///方法2的优化，新建这个公用的方法，其他地方想用就比较方便
var tryFun = function(promise){ 
  return promise.then(res=>[res,null]).catch(err=>[null,err]); 
}

var run = async function()
{   
    //同时获取成功和失败的值
    var [res,err] =  await tryFun(test2fun)
    //这里没法执行，会报错
    console.log("res:"+res);
    console.log("err:"+err); 
}

# 二、组件

# 1、创建

根目录下创建文件夹 components-->创建组件文件夹headCom-->右击创建组件headcomponent

# 2、引用

2.1、在需要引用界面所在的json里面配置引用

"usingComponents": {
  "headerCom":"/components/headCom/headcomponent"
}

2.2、 wxml里面直接使用

<headerCom></headerCom>

# 3、定制参数

3.1、配置组件属性

在组件的js文件夹中配置属性内容

/**
 * 组件的属性列表
 */
properties: {
  titleStr:
  {
    type:String,
    value:"标题",
  },
  content:
  {
    type:String,
    value:"内容内容内容内容内容"
  }
},

3.2、使用组件属性

<view class="header">
  <view class="big">
    {{titleStr}}
  </view> 
  <view class="small">
    {{content}}
  </view>
</view>

3.3、传递参数

引用界面可以传递自己的参数了

<headerCom titleStr="首页Item1" content="首页内容11111"></headerCom>

# 三、使用sass

1、vscode中安装easysass

2、导入已安装的vscode扩展

微信开发者工具 视图=>扩展=>右侧三点=>导入已安装的vscode扩展

3、微信开发者工具 编辑=>打开编辑器扩展目录=>进入sass=>编辑package.json文件=>内容如下

"easysass.formats": {
					"type": "array",
					"default": [
						{
							"format": "expanded",
							"extension": ".wxss"
						} 
					],
					"description": "Define format(s) for outputted css files. Use \"nested\", \"expanded\", \"compact\" or \"compressed\" as a format."
				},

4、重新启动，微信开发者工具

5、使用

​ 5.1、在界面文件夹下创建index.scss

​ 5.2、编辑scss文件，就会自动编写对应的wxcss

例：

.header{
  padding: 10rpx;
  text-align: center; 
  background-color: green;
  .big{ 
    font-size: 40rpx;
    margin-bottom: 10rpx;
  }
  .small{
    font-size: 30rpx;
  }
}
